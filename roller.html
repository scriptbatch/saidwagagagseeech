Hereâ€™s 1 free Apple Point to keep playing."); } // Update user data users[username].applePoints = applePoints; localStorage.setItem("users", JSON.stringify(users)); // Update UI updateApplePoints(); alert(`You won ${winnings} Apple Points!`); } });
